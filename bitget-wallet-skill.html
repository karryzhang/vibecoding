Quick Start

Get your AI agent on-chain in minutes.

# Install the package npm install bitget-wallet-skill
import { BitgetWalletSkill } from 'bitget-wallet-skill'; // Initialize const skill = new BitgetWalletSkill({ apiKey: process.env.BITGET_WALLET_API_KEY, }); // Get a swap quote (ETH → USDC on Ethereum) const quote = await skill.swapQuote({ chainId: '1', fromToken: '0xEeeeeEeeeEeEeeEeEeEeeEEEeeeeEeeeeeeeEEeE', toToken: '0xA0b86991c6218b36c1d19D4a2e9Eb0cE3606eB48', amount: '1000000000000000000', // 1 ETH slippage: '0.5', }); // Execute the swap const result = await skill.swapSend({ ...quote, userAddress: '0xYourAddress', }); console.log('Tx Hash:', result.txHash);
// Cross-chain: USDC on Base → USDT on BNB Chain (Gasless!) const orderQuote = await skill.orderQuote({ fromChainId: '8453', // Base toChainId: '56', // BNB Chain fromToken: '0x833589fCD6eDb6E08f4c7C32D4f71b54bdA02913', // USDC toToken: '0x55d398326f99059fF775485246999027B3197955', // USDT amount: '100000000', // 100 USDC gasless: true, // Zero gas required! }); // Create and submit the order const order = await skill.orderCreate(orderQuote); const signed = await signer.signTransaction(order.txData); const status = await skill.orderSubmit({ orderId: order.id, signature: signed }); console.log('Order:', status.orderId, 'Status:', status.status);
